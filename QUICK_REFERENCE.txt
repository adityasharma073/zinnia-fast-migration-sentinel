{
  "extraction_timestamp": "2026-03-27T11:43:46.439444Z",
  "pdf": {
    "status": "success|error",
    "content": "# Title\n...",      ← Markdown with structure
    "metadata": { ... },              ← Page count, etc.
    "tables": [ ... ]                 ← Extracted tables
  },
  "xml": {
    "status": "success|error",
    "data": { ... },                  ← Full parsed XML
    "metadata": { ... }               ← File size, encoding
  },
  "summary": {
    "overall_status": "success|partial_failure",
    "pdf_status": "...",
    "xml_status": "..."
  }
}

Key Fields:
  status: "success" (data extracted) or "error" (with message)
  content: Full PDF as Markdown (preserves structure)
  data: XML as Python dictionary (nested structure)
  timestamp: ISO-8601 UTC time

🔧 API REFERENCE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Main Function:
  extract_and_normalize(pdf_path, xml_path)
  → Returns combined JSON with both extractions

Class Methods:
  DocumentExtractor.extract_pdf(pdf_path)
  → Returns PDF extraction results

  DocumentExtractor.extract_xml(xml_path)
  → Returns XML extraction results

Result Structure:
  result['status'] → 'success' or 'error'
  result['data'] → extracted data
  result['error'] → error message (if failed)
  result['metadata'] → file info
